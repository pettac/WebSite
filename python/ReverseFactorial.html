Reverse Factorial

Solution:

def factorial(number):
    i = 2
    while number % i == 0:
        number /= i
        i += 1
        if number/i == 1:
            break
    else:
        return 0
    return number


def main():
    value = input("Enter a number: ")
    result = factorial(int(value))
    if result == 0:
        print("NONE")
    else:
        print(str(int(result))+"!")


if __name__ == "__main__":
    main()
