Game of Threes

Solution:

def three(num):
    while num != 1:
        if num % 3 == 0:
            print(int(num), 0)
        elif num % 3 == 1:
            print(int(num), -1)
            num -= 1
        elif num % 3 == 2:
            print(int(num), 1)
            num += 1
        num /= 3
    print(int(num))

if __name__ == "__main__":
    three(31337357)
