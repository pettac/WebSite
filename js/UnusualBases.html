Fibonacci Sequence

function fib(n){
    var a = 1;
    var b = 1;
    var temp;
    for (var i=0; i<n-1; i++){
        temp = a;
        a = b;
        b = temp + b;
    }
    return a;
}

Binary to Decimal

function binaryToDecimal(num){
    num = num.split("").reverse().join("");

    var count = 0.0;
    var pos = 0;
    for (var i=0; i<num.length; i++){
        if(num[i] == "1")
            count += Math.pow(2, pos);
        pos += 1;
    }
    return count.toString();
}

Decimal To Binary

function decimalToBinary(num){
    var result = "";
    while (num != 0){
        if (num % 2 == 0)
            result += "0";
        else{
            result += "1";
        }
        num = Math.floor(num/2);
    }
    return result.split("").reverse().join("");
}

Binary to Fibonacci

function binaryToFib(num) {
    num = num.split("").reverse().join("");
    var pos = 1;
    var count = 0;
    for (var i=0, len=num.length; i<len; i++) {
        if (num[i] == "1") {
            count += fib(pos);
        }
        pos += 1;
    }
    return count;
}
