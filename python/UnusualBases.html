def todeca(num):
    num = str(num)
    num = num[::-1]
    count = 0
    pos = 0
    for i in num:
        if i == '1':
            count += 2**pos
        pos += 1
    return count


def tobinary(num):
    result = ""
    while num != 0:
        if num % 2 == 0:
            result += '0'
        else:
            result += '1'
        num //= 2
    result = result[::-1]
    return result


def fib(n):
    a, b = 1,1
    for i in range(n-1):
        a, b = b, a+b
    return int(a)


def tofib(num):
    num = str(num)
    num = num[::-1]
    pos = 1
    count = 0
    for i in num:
        if i == '1':
            count += int(fib(pos))
        pos += 1
    return count
