function gameOfThrees(){
    var results = [];
    var steps = [];
    var input = document.getElementById("input").value;
    var num = parseInt(input);
    while (num != 1){
        if (num % 3 == 0) {
            results.push(num);
            steps.push(0);
        }
        else if (num % 3 == 1){
            results.push(num);
            steps.push(-1);
            num -= 1;
        }
        else if (num % 3 == 2){
            results.push(num);
            steps.push(1);
            num += 1
        }
        num /= 3;
    }
    var text = "<ul>";
    for (var i in results){
        text += "<li>" + results[i].toString() + " " + steps[i].toString()+"</li>";
    }
    text += "<li>1</li>";
    text += "</ul>";
    document.getElementById("result").innerHTML = text;
}
