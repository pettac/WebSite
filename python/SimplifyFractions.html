Simplifying Fractions

Solution:

def simplify(x, y):
    while y != 0:
        x, y = y, x % y
    return x


def main():
    x = 4
    y = 8
    gcd = simplify(x, y)
    x /= gcd
    y /= gcd
    print(int(x), int(y))


if __name__ == "__main__":
    main()
