function gcd() {
    var x = parseInt(document.getElementById("numerator").value);
    var y = parseInt(document.getElementById("denominator").value);
    var num = x;
    var den = y;
    var temp;
    while (y != 0) {
        temp = x;
        x = y;
        y = temp % y;
    }
    num /= x;
    den /= x;
    document.getElementById("result").innerHTML = "<sup>"+num+"</sup>&frasl;<sub>"+den+"</sub>";
}
