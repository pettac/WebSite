function factorial() {
    var i = 2;
    var input = document.getElementById("input").value;
    var num = parseInt(input)
    while (num % i == 0) {
        num /= i;
        i += 1;
        if (num / i == 1) {
            document.getElementById("result").innerHTML = input + " is "+num.toString()+"!";
            break;
        }
    }
    if (num % 0 != 0 && num/i != 1){
        document.getElementById("result").innerHTML = "No factorial found";
    }
}
